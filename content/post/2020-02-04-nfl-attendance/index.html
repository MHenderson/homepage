library(tidyverse)

attendance <- read_csv('https://raw.githubusercontent.com/rfordatascience/tidytuesday/master/data/2020/2020-02-04/attendance.csv')

standings <- read_csv('https://raw.githubusercontent.com/rfordatascience/tidytuesday/master/data/2020/2020-02-04/standings.csv')

games <- read_csv('https://raw.githubusercontent.com/rfordatascience/tidytuesday/master/data/2020/2020-02-04/games.csv')

attendance %>%
  left_join(standings, by = c("year", "team_name", "team")) %>%
  mutate(team = paste(team, team_name)) %>%
  filter(year==2003) %>%
  ggplot(aes(week, weekly_attendance)) +
  geom_step() +
  geom_smooth() +
  facet_wrap(~team)
